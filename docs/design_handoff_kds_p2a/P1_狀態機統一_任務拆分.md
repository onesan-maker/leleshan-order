## 2. P1-A｜建立 `order-state-machine.js` 唯一真相

### 目標
把所有狀態定義（label、tone、合法轉移、是否進 KDS、是否進取餐看板、是否通知顧客）集中到一個檔案，作為**唯一 source of truth**。

### 修改檔案
- ✏️ 新增：`order-state-machine.js`
- ✏️ 修改：`index.html`、`kds.html`、`pos.html`、`pos-admin.html`、`admin.html`、`packing.html`、`pickup-board.html`、`admin-orders.html`（在 `order-helpers.js` 之前載入）

### 完整檔案內容

新增檔案 `order-state-machine.js`：

```js
(function () {
  // ============================================================
  // 樂樂山訂單狀態機 — 唯一 source of truth
  // 任何顯示、轉移、過濾邏輯都應透過此模組
  // ============================================================

  // 正式狀態定義
  var STATES = {
    pending_confirmation: {
      label: "待確認",
      tone: "warn",
      visibleOnKds: true,
      visibleOnPickupBoard: false,
      notifyCustomer: false,        // 由 received 通知處理
      column: "pending"
    },
    accepted: {
      label: "已接單",
      tone: "accepted",
      visibleOnKds: true,
      visibleOnPickupBoard: true,
      notifyCustomer: true,
      column: "preparing"
    },
    preparing: {
      label: "製作中",
      tone: "cooking",
      visibleOnKds: true,
      visibleOnPickupBoard: true,
      notifyCustomer: false,
      column: "preparing"
    },
    ready: {
      label: "可取餐",
      tone: "ready",
      visibleOnKds: true,
      visibleOnPickupBoard: true,
      notifyCustomer: true,
      column: "ready"
    },
    completed: {
      label: "已完成",
      tone: "picked",
      visibleOnKds: false,
      visibleOnPickupBoard: false,
      notifyCustomer: false,
      column: null
    },
    cancelled: {
      label: "已取消",
      tone: "cancelled",
      visibleOnKds: false,
      visibleOnPickupBoard: false,
      notifyCustomer: true,
      column: null
    }
  };

  // 合法轉移（from → [to, to...]）
  var TRANSITIONS = {
    pending_confirmation: ["accepted", "cancelled"],
    accepted:             ["preparing", "cancelled"],
    preparing:            ["ready", "cancelled"],
    ready:                ["completed", "cancelled"],
    completed:            [],
    cancelled:            []
  };

  // 舊狀態映射（讀取時 normalize；寫入禁用）
  var LEGACY_MAP = {
    new:       "pending_confirmation",  // ⚠ 重要：舊系統 new = 待確認
    cooking:   "preparing",
    packing:   "preparing",
    done:      "ready",
    picked_up: "completed",
    canceled:  "cancelled"
  };

  // 來源標籤
  var SOURCE_LABELS = {
    line:      "LINE點餐",
    liff:      "LINE點餐",   // legacy
    pos:       "現場",
    walk_in:   "現場",       // legacy
    onsite:    "現場",       // legacy
    phone:     "電話",
    ubereats:  "UberEats",
    foodpanda: "Foodpanda",
    manual:    "人工建立"
  };

  // KDS 短標籤（卡片右上 badge）
  var SOURCE_SHORT_LABELS = {
    line: "LINE", liff: "LINE",
    pos: "現場", walk_in: "現場", onsite: "現場",
    phone: "電話",
    ubereats: "UE", foodpanda: "FP",
    manual: "人工"
  };

  // ──────────── 對外 API ────────────

  function normalize(status) {
    var key = String(status || "").trim().toLowerCase();
    if (!key) return "pending_confirmation";
    return LEGACY_MAP[key] || key;
  }

  function meta(status) {
    var key = normalize(status);
    return STATES[key] || STATES.pending_confirmation;
  }

  function label(status) {
    return meta(status).label;
  }

  function tone(status) {
    return meta(status).tone;
  }

  function isVisibleOnKds(status) {
    return meta(status).visibleOnKds;
  }

  function isVisibleOnPickupBoard(status) {
    return meta(status).visibleOnPickupBoard;
  }

  function shouldNotifyCustomer(status) {
    return meta(status).notifyCustomer;
  }

  function column(status) {
    return meta(status).column;
  }

  // 是否可從 from 轉到 to
  function canAdvance(from, to) {
    var nFrom = normalize(from);
    var nTo = normalize(to);
    return (TRANSITIONS[nFrom] || []).indexOf(nTo) >= 0;
  }

  // 嚴格驗證；不合法時 throw
  function assertAdvance(from, to) {
    if (!canAdvance(from, to)) {
      throw new Error("非法狀態轉移：" + from + " → " + to);
    }
  }

  // 來源標籤
  function sourceLabel(source) {
    return SOURCE_LABELS[String(source || "").toLowerCase()] || "現場";
  }

  function sourceShortLabel(source) {
    return SOURCE_SHORT_LABELS[String(source || "").toLowerCase()] || "現場";
  }

  // KDS 全部可見狀態
  function kdsVisibleStatuses() {
    return Object.keys(STATES).filter(function (s) { return STATES[s].visibleOnKds; });
  }

  // 取餐看板可見狀態
  function pickupBoardStatuses() {
    return Object.keys(STATES).filter(function (s) { return STATES[s].visibleOnPickupBoard; });
  }

  // KDS 三欄
  function kdsColumns() {
    return [
      { key: "pending",    label: "待確認", statuses: ["pending_confirmation"] },
      { key: "preparing",  label: "待/製作中", statuses: ["accepted", "preparing"] },
      { key: "ready",      label: "可取餐", statuses: ["ready"] }
    ];
  }

  window.OrderStateMachine = {
    STATES: STATES,
    TRANSITIONS: TRANSITIONS,
    normalize: normalize,
    meta: meta,
    label: label,
    tone: tone,
    isVisibleOnKds: isVisibleOnKds,
    isVisibleOnPickupBoard: isVisibleOnPickupBoard,
    shouldNotifyCustomer: shouldNotifyCustomer,
    column: column,
    canAdvance: canAdvance,
    assertAdvance: assertAdvance,
    sourceLabel: sourceLabel,
    sourceShortLabel: sourceShortLabel,
    kdsVisibleStatuses: kdsVisibleStatuses,
    pickupBoardStatuses: pickupBoardStatuses,
    kdsColumns: kdsColumns
  };
})();
```

### HTML 載入順序（每個會用到狀態的頁面）

在 `<script src="order-helpers.js">` **之前**插入：

5. 沒有任何現有頁面壞掉（這個檔只是新增 global，未取代任何邏輯）

## 7. P1-F｜`firestore.rules` 收緊規則

### 目標
1. 限定 source 對應的允許 status：LINE 來源只能寫 `pending_confirmation`，POS 來源只能寫 `pending_confirmation` 或 `accepted`。
2. 移除 `'new'` 在 create 規則的允許（避免新訂單寫入 deprecated 狀態）。
3. 訂單轉移時，rules 不檢查合法性（交給應用層 + state machine），但檢查 storeId 不可變更。

### 修改檔案
- ✏️ `firestore.rules`（orders match block）

### 具體 Diff

**找到 `match /orders/{orderId}` 的 `allow create`：**
```ruby
# ❌ 原本
allow create: if request.resource.data.storeId is string
  && request.resource.data.status in ['new', 'pending_confirmation']
  && request.resource.data.source in ['liff', 'onsite', 'pos', 'manual', 'ubereats', 'foodpanda', 'line', 'walk_in', 'phone']
  || (isAdmin() && ...);

# ✅ 改為
allow create: if request.resource.data.storeId is string
  && (
    // LINE 來源：只能 pending_confirmation
    (request.resource.data.source in ['liff', 'line']
      && request.resource.data.status == 'pending_confirmation')
    ||
    // POS 來源：可 pending_confirmation 或 accepted
    (request.resource.data.source in ['pos', 'onsite', 'walk_in']
      && request.resource.data.status in ['pending_confirmation', 'accepted'])
    ||
    // 電話 / 平台 / 人工
    (request.resource.data.source in ['phone', 'manual', 'ubereats', 'foodpanda']
      && request.resource.data.status in ['pending_confirmation', 'accepted'])
  )
  || (isAdmin()
      && request.resource.data.storeId is string
      && canManage(request.resource.data.storeId)
      && request.resource.data.isTest == true);
```

**`update` 規則加上 storeId 不可變更：**

```ruby
allow update: if (canReadOrders(resource.data.storeId)
                  && request.resource.data.storeId == resource.data.storeId)
  || (resource.data.lineUserId is string
      && resource.data.lineUserId.size() > 0
      && request.resource.data.lineUserId == resource.data.lineUserId
      && (
        (resource.data.status == 'pending_confirmation'
          && (request.resource.data.status == 'cancelled'
              || request.resource.data.customerHidden == true))
        || request.resource.data.archived == true
      ));
```

### 驗收
1. 部署規則：`firebase deploy --only firestore:rules`
2. 在 console 嘗試 `db.collection("orders").add({source:"line", status:"accepted", storeId:"store_1"})` → 應被拒絕
3. LIFF 正常下單成功（status=pending_confirmation）
4. POS 下單成功（status=accepted）
5. 顧客可取消自己 pending_confirmation 的訂單

### 回滾
deploy 前的舊版 rules（建議先 `firebase firestore:rules:get` 備份）

### 風險提醒
此任務有潛在 break change：若有其他客戶端硬寫 `status: "new"` 來建立訂單，會被擋下。**部署前需 grep 全專案** `status.*new`、`status: 'new'` 確認沒有遺漏。
